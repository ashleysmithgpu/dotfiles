= airline =

Install:

```
mkdir ~/.config/nvim
cd ~/.config/nvim
git clone https://github.com/vim-airline/vim-airline.git
cp -R vim-airline/plugin vim-airline/autoload .
rm -r vim-airline
cd ~/github/
git clone https://github.com/tmux-plugins/tmux-resurrect
```

= disable back button on mouse =
/etc/udev/rules.d/10-custom-mouse.rules


ATTRS{name}=="Kingsis Peripherals  Evoluent VerticalMouse 3", ACTION=="add", RUN+="/bin/sh -c \"export DISPLAY=:0;export XAUTHORITY=/home/ash/.Xauthority;xinput --list | grep 'Kingsis Peripherals  Evoluent VerticalMouse 3' | grep -o 'id=[0-9]*' | cut -f2- -d= | while read id; do xinput set-button-map $id 1 2 3 4 5 6 7 0; done\""







= icons/launchers for ubuntu =

Version=1.0
Comment=
Type=Application
Terminal=false
Exec=/usr/bin/p4v
Name=p4v
Icon=p4v
StartupNotify=true
