
= install =

```
sudo add-apt-repository ppa:neovim-ppa/stable
sudo apt update
sudo apt install neovim
sudo apt install fish tmux
```

Set neovim as default:

```
sudo update-alternatives --install /usr/bin/vim vim /usr/bin/nvim 60
```

= fonts =

```
sudo apt install powerline
```

= airline =

Install:

```
cd ~/.config/nvim
git clone https://github.com/vim-airline/vim-airline.git
cp -R vim-airline/plugin vim-airline/autoload .
rm -r vim-airline
```

= configs =

```
mkdir ~/.config/nvim/
ln /home/ash/dotfiles/vim/.vimrc ~/.config/nvim/init.vim
mkdir ~/.config/fish/
ln /home/ash/dotfiles/fish/config.fish ~/.config/fish/config.fish
ln /home/ash/dotfiles/tmux/.tmux.conf ~/.tmux.conf
```

= syntax =

```
cd ~/.config/nvim
mkdir syntax
cd syntax
wget https://raw.githubusercontent.com/rust-lang/rust.vim/master/syntax/rust.vim
```
